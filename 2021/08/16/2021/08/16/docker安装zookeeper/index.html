docker安装zookeeper

#拉取镜像
docker pull zookeeper

#创建挂载文件夹
mkdir -p /opt/zook

#启动容器
docker run -d -v /opt/zook:/data -v /opt/zook:/datalog  -e ZOO_MY_ID=1 -e ZOO_SERVERS='server.1=172.110.110.101:2888:3888'  --net=host --name zookeeper --privileged zookeeper

#设置容器开机自启
docker update zookeeper --restart=always
